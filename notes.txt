We are going to import our completed site and arrange the files in a format
that WP likes... 

Need to use to link to our style sheets in functions.php. All of our style links will be included in here (including external ones)

<?php
function nhah_theme_styles(){
	wp_enqueue_stlye('normalize_css', get_template_directory_uri() . '/css/normalize.css');
}
add_action('wp_enqueue_scripts', 'wpt_theme_styles')

function wpt_theme_js(){
	
	wp_enqueue_script('main_js', get_template_directory() . '/js/main.js', '', '', false );
}
add_action('wp_enqueue_scripts', 'wpt_theme_js')


?>
